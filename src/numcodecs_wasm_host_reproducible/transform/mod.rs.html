13pub fn transform_wasm_component(wasm_component: impl Into<Vec<u8>>) -> Result<Vec<u8>, Error> {
14    let NumcodecsWitInterfaces {
15        package,
16        codec: codec_interface,
17        types: types_interface,
18        ..
19    } = NumcodecsWitInterfaces::get();
20
21    // create a new WAC composition graph with the WASI component packages
22    //  pre-registered and the numcodecs:wasm/perf interface pre-exported
23    let PreparedCompositionGraph {
24        graph: wac,
25        wasi: wasi_component_packages,
26    } = get_prepared_composition_graph()?;
27    let mut wac = wac.clone();
28
29    // parse and instantiate the root package, which exports numcodecs:abc/codec
30    let numcodecs_codec_package = wac_graph::types::Package::from_bytes(
31        &format!("{}", package.name()),
32        package.version(),
33        wasm_component,
34        wac.types_mut(),
35    )?;
36
37    let numcodecs_codec_world = &wac.types()[numcodecs_codec_package.ty()];
38    let numcodecs_codec_imports = extract_component_ports(&numcodecs_codec_world.imports)?;
39
40    let numcodecs_codec_package = wac.register_package(numcodecs_codec_package)?;
41    let numcodecs_codec_instance = wac.instantiate(numcodecs_codec_package);
42
43    // list the imports that the linker will provide
44    let linker_provided_imports = [
45        &WasiSandboxedStdioInterface::get().stdio,
46        &WasiLoggingInterface::get().logging,
47    ];
48    // trivial imports that require no linking
49    let trivial_imports = [types_interface];
50
51    // initialise the unresolved imports to the imports of the root package
52    let mut unresolved_imports = VecMap::new();
53    for import in &numcodecs_codec_imports {
54        unresolved_imports
55            .entry(import.clone())
56            .or_insert_with(Vec::new)
57            .push(numcodecs_codec_instance);
58    }
59
60    // track all non-root instances, which may fulfil imports
61    let mut package_instances = VecMap::new();
62
63    // iterate while not all unresolved imports have been satisfied
64    while let Some((unresolved_import, dependents)) = unresolved_imports.pop() {
65        // some imports are trivial and require no linking
66        if trivial_imports.contains(&&unresolved_import) {
67            continue;
68        }
69
70        // some imports will be provided by the linker
71        if linker_provided_imports.contains(&&unresolved_import) {
72            continue;
73        }
74
75        // resolve the import
76        // - either it has been instantiated already
77        // - or we instantiate the WASI component that will provide it
78        // - or it is unresolvable and we raise an error
79        let dependency_instance =
80            if let Some(dependency_instance) = package_instances.get(unresolved_import.package()) {
81                *dependency_instance
82            } else if let Some(component_package) = wasi_component_packages
83                .iter()
84                .find(|component_package| component_package.exports.contains(&unresolved_import))
85            {
86                let PackageWithPorts {
87                    package: component_package,
88                    imports: component_imports,
89                    exports: component_exports,
90                } = component_package;
91
92                // instantiate the component package
93                let component_instance = wac.instantiate(*component_package);
94
95                // require all imports of the component package to be resolved
96                for import in component_imports {
97                    unresolved_imports
98                        .entry(import.clone())
99                        .or_insert_with(Vec::new)
100                        .push(component_instance);
101                }
102
103                for export in component_exports {
104                    // register this instance's package so that its exports can later
105                    //  fulfil more imports
106                    package_instances.insert(export.package().clone(), component_instance);
107                }
108
109                component_instance
110            } else {
111                return Err(anyhow!(
112                    "WASM component requires unresolved import {unresolved_import}"
113                ));
114            };
115
116        let import_str = &format!("{unresolved_import}");
117        let dependency_export = wac.alias_instance_export(dependency_instance, import_str)?;
118        for dependent in dependents {
119            wac.set_instantiation_argument(dependent, import_str, dependency_export)?;
120        }
121    }
122
123    assert!(unresolved_imports.into_vec().is_empty());
124
125    // export the numcodecs:abc/codec interface
126    let numcodecs_codecs_str = &format!("{codec_interface}");
127    let numcodecs_codecs_export =
128        wac.alias_instance_export(numcodecs_codec_instance, numcodecs_codecs_str)?;
129    wac.export(numcodecs_codecs_export, numcodecs_codecs_str)?;
130
131    // encode the WAC composition graph into a WASM component and validate it
132    let wasm = wac.encode(wac_graph::EncodeOptions {
133        define_components: true,
134        // we do our own validation right below
135        validate: false,
136        processor: None,
137    })?;
138
139    wasmparser::Validator::new_with_features(
140        wasmparser::WasmFeaturesInflated {
141            // MUST: float operations are required
142            //       (and our engine's transformations makes them deterministic)
143            floats: true,
144            // MUST: codecs and reproducible WASI are implemented as components
145            component_model: true,
146            ..crate::engine::DETERMINISTIC_WASM_MODULE_FEATURES
147        }
148        .into(),
149    )
150    .validate_all(&wasm)?;
151
152    Ok(wasm)
153}

155struct PreparedCompositionGraph {
156    graph: wac_graph::CompositionGraph,
157    wasi: Box<[PackageWithPorts]>,
158}

160fn get_prepared_composition_graph() -> Result<&'static PreparedCompositionGraph, Error> {
161    static PREPARED_COMPOSITION_GRAPH: OnceLock<Result<PreparedCompositionGraph, Error>> =
162        OnceLock::new();
163
164    let prepared_composition_graph = PREPARED_COMPOSITION_GRAPH.get_or_init(|| {
165        let PerfWitInterfaces {
166            perf: perf_interface,
167            ..
168        } = PerfWitInterfaces::get();
169
170        // create a new WAC composition graph
171        let mut wac = wac_graph::CompositionGraph::new();
172
173        // parse and register the WASI component packages
174        let wasi_component_packages =
175            register_wasi_component_packages(&mut wac)?.into_boxed_slice();
176
177        // create, register, and instantiate the numcodecs:wasm package
178        let numcodecs_wasm_perf_instance = instantiate_numcodecs_wasm_perf_package(&mut wac)?;
179
180        // export the numcodecs:wasm/perf interface
181        let numcodecs_wasm_perf_str = &format!("{perf_interface}");
182        let numcodecs_wasm_perf_export =
183            wac.alias_instance_export(numcodecs_wasm_perf_instance, numcodecs_wasm_perf_str)?;
184        wac.export(numcodecs_wasm_perf_export, numcodecs_wasm_perf_str)?;
185
186        Ok(PreparedCompositionGraph {
187            graph: wac,
188            wasi: wasi_component_packages,
189        })
190    });
191
192    match prepared_composition_graph {
193        Ok(prepared_composition_graph) => Ok(prepared_composition_graph),
194        Err(err) => Err(anyhow!(err)),
195    }
196}

198struct PackageWithPorts {
199    package: wac_graph::PackageId,
200    imports: Box<[wasm_component_layer::InterfaceIdentifier]>,
201    exports: Box<[wasm_component_layer::InterfaceIdentifier]>,
202}
203
204fn register_wasi_component_packages(
205    wac: &mut wac_graph::CompositionGraph,
206) -> Result<Vec<PackageWithPorts>, Error> {
207    const WASI_COMPONENTS: &[(&str, &[u8])] = &[(
208        "wasi-sandboxed:merged",
209        wasi_sandboxed_component_provider::MERGED_COMPONENT,
210    )];
211
212    let wasi_component_packages = WASI_COMPONENTS
213        .iter()
214        .map(|(component_name, component_bytes)| -> Result<_, Error> {
215            let component_package = wac_graph::types::Package::from_bytes(
216                component_name,
217                None,
218                Vec::from(*component_bytes),
219                wac.types_mut(),
220            )?;
221
222            let component_world = &wac.types()[component_package.ty()];
223
224            let component_imports = extract_component_ports(&component_world.imports)?;
225            let component_exports = extract_component_ports(&component_world.exports)?;
226
227            let component_package = wac.register_package(component_package)?;
228
229            Ok(PackageWithPorts {
230                package: component_package,
231                imports: component_imports.into_boxed_slice(),
232                exports: component_exports.into_boxed_slice(),
233            })
234        })
235        .collect::<Result<Vec<_>, _>>()?;
236
237    Ok(wasi_component_packages)
238}
239
240fn extract_component_ports(
241    ports: &indexmap::IndexMap<String, wac_graph::types::ItemKind>,
242) -> Result<Vec<wasm_component_layer::InterfaceIdentifier>, anyhow::Error> {
243    ports
244        .iter()
245        .filter_map(|(import, kind)| match kind {
246            wac_graph::types::ItemKind::Instance(_) => Some(
247                wasm_component_layer::InterfaceIdentifier::try_from(import.as_str()),
248            ),
249            _ => None,
250        })
251        .collect::<Result<Vec<_>, _>>()
252}

254fn instantiate_numcodecs_wasm_perf_package(
255    wac: &mut wac_graph::CompositionGraph,
256) -> Result<wac_graph::NodeId, Error> {
257    let PerfWitInterfaces {
258        perf: perf_interface,
259        ..
260    } = PerfWitInterfaces::get();
261
262    // create, register, and instantiate the numcodecs:wasm/perf package
263    let numcodecs_wasm_perf_package = wac_graph::types::Package::from_bytes(
264        &format!("{}", perf_interface.package().name()),
265        perf_interface.package().version(),
266        create_numcodecs_wasm_perf_component()?,
267        wac.types_mut(),
268    )?;
269
270    let numcodecs_wasm_perf_package = wac.register_package(numcodecs_wasm_perf_package)?;
271    let numcodecs_wasm_perf_instance = wac.instantiate(numcodecs_wasm_perf_package);
272
273    Ok(numcodecs_wasm_perf_instance)
274}

276fn create_numcodecs_wasm_perf_component() -> Result<Vec<u8>, Error> {
277    const ROOT: &str = "root";
278
279    let PerfWitInterfaces {
280        perf: perf_interface,
281        instruction_counter,
282    } = PerfWitInterfaces::get();
283
284    let mut module = create_numcodecs_wasm_perf_module();
285
286    let mut resolve = wit_parser::Resolve::new();
287
288    let interface = resolve.interfaces.alloc(wit_parser::Interface {
289        name: Some(String::from(perf_interface.name())),
290        types: indexmap::IndexMap::new(),
291        #[expect(clippy::iter_on_single_items)]
292        functions: [(
293            String::from(instruction_counter),
294            wit_parser::Function {
295                name: String::from(instruction_counter),
296                kind: wit_parser::FunctionKind::Freestanding,
297                params: Vec::new(),
298                result: Some(wit_parser::Type::U64),
299                docs: wit_parser::Docs { contents: None },
300                stability: wit_parser::Stability::Unknown,
301            },
302        )]
303        .into_iter()
304        .collect(),
305        docs: wit_parser::Docs { contents: None },
306        package: None, // The package is linked up below
307        stability: wit_parser::Stability::Unknown,
308    });
309
310    let package_name = wit_parser::PackageName {
311        namespace: String::from(perf_interface.package().name().namespace()),
312        name: String::from(perf_interface.package().name().name()),
313        version: perf_interface.package().version().cloned(),
314    };
315    let package = resolve.packages.alloc(wit_parser::Package {
316        name: package_name.clone(),
317        docs: wit_parser::Docs { contents: None },
318        #[expect(clippy::iter_on_single_items)]
319        interfaces: [(String::from(perf_interface.name()), interface)]
320            .into_iter()
321            .collect(),
322        worlds: indexmap::IndexMap::new(),
323    });
324    resolve.package_names.insert(package_name, package);
325
326    if let Some(interface) = resolve.interfaces.get_mut(interface) {
327        interface.package = Some(package);
328    }
329
330    let world = resolve.worlds.alloc(wit_parser::World {
331        name: String::from(ROOT),
332        imports: indexmap::IndexMap::new(),
333        #[expect(clippy::iter_on_single_items)]
334        exports: [(
335            wit_parser::WorldKey::Interface(interface),
336            wit_parser::WorldItem::Interface {
337                id: interface,
338                stability: wit_parser::Stability::Unknown,
339            },
340        )]
341        .into_iter()
342        .collect(),
343        package: None, // The package is linked up below
344        docs: wit_parser::Docs { contents: None },
345        includes: Vec::new(),
346        include_names: Vec::new(),
347        stability: wit_parser::Stability::Unknown,
348    });
349
350    let root_name = wit_parser::PackageName {
351        namespace: String::from(ROOT),
352        name: String::from("component"),
353        version: perf_interface.package().version().cloned(),
354    };
355    let root = resolve.packages.alloc(wit_parser::Package {
356        name: root_name.clone(),
357        docs: wit_parser::Docs { contents: None },
358        interfaces: indexmap::IndexMap::new(),
359        #[expect(clippy::iter_on_single_items)]
360        worlds: [(String::from(ROOT), world)].into_iter().collect(),
361    });
362    resolve.package_names.insert(root_name, root);
363
364    if let Some(world) = resolve.worlds.get_mut(world) {
365        world.package = Some(root);
366    }
367
368    wit_component::embed_component_metadata(
369        &mut module,
370        &resolve,
371        world,
372        wit_component::StringEncoding::UTF8,
373    )?;
374
375    let mut encoder = wit_component::ComponentEncoder::default()
376        .module(&module)
377        .context("wit_component::ComponentEncoder::module failed")?;
378
379    let component = encoder
380        .encode()
381        .context("wit_component::ComponentEncoder::encode failed")?;
382
383    Ok(component)
384}

386fn create_numcodecs_wasm_perf_module() -> Vec<u8> {
387    let PerfWitInterfaces {
388        perf: perf_interface,
389        instruction_counter,
390    } = PerfWitInterfaces::get();
391
392    let mut module = wasm_encoder::Module::new();
393
394    // Encode the type section with
395    //  types[0] = () -> i64
396    let mut types = wasm_encoder::TypeSection::new();
397    let ty0 = types.len();
398    types.ty().function([], [wasm_encoder::ValType::I64]);
399    module.section(&types);
400
401    // Encode the function section with
402    //  functions[0] = fn() -> i64 [ types[0] ]
403    let mut functions = wasm_encoder::FunctionSection::new();
404    let fn0 = functions.len();
405    functions.function(ty0);
406    module.section(&functions);
407
408    // Encode the export section with
409    //  {perf_interface}#{instruction_counter} = functions[0]
410    let mut exports = wasm_encoder::ExportSection::new();
411    exports.export(
412        &format!("{perf_interface}#{instruction_counter}"),
413        wasm_encoder::ExportKind::Func,
414        fn0,
415    );
416    module.section(&exports);
417
418    // Encode the code section.
419    let mut codes = wasm_encoder::CodeSection::new();
420    let mut fn0 = wasm_encoder::Function::new([]);
421    fn0.instruction(&wasm_encoder::Instruction::Unreachable);
422    fn0.instruction(&wasm_encoder::Instruction::End);
423    codes.function(&fn0);
424    module.section(&codes);
425
426    // Extract the encoded WASM bytes for this module
427    module.finish()
428}
