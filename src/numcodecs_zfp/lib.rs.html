33/// Codec providing compression using ZFP
34pub struct ZfpCodec {
35    /// ZFP compression mode
36    pub mode: ZfpCompressionMode,
37}
38
39#[derive(Clone, Debug, Serialize, Deserialize, JsonSchema)]
40#[serde(tag = "mode")]
41#[serde(deny_unknown_fields)]
42/// ZFP compression mode
43pub enum ZfpCompressionMode {
44    #[serde(rename = "expert")]
45    /// The most general mode, which can describe all four other modes
46    Expert {
47        /// Minimum number of compressed bits used to represent a block
48        min_bits: u32,
49        /// Maximum number of bits used to represent a block
50        max_bits: u32,
51        /// Maximum number of bit planes encoded
52        max_prec: u32,
53        /// Smallest absolute bit plane number encoded.
54        ///
55        /// This parameter applies to floating-point data only and is ignored
56        /// for integer data.
57        min_exp: i32,
58    },
59    /// In fixed-rate mode, each d-dimensional compressed block of `$4^d$`
60    /// values is stored using a fixed number of bits. This number of
61    /// compressed bits per block is amortized over the `$4^d$` values to give
62    /// a rate of `$rate = \frac{maxbits}{4^d}$` in bits per value.
63    #[serde(rename = "fixed-rate")]
64    FixedRate {
65        /// Rate in bits per value
66        rate: f64,
67    },
68    /// In fixed-precision mode, the number of bits used to encode a block may
69    /// vary, but the number of bit planes (the precision) encoded for the
70    /// transform coefficients is fixed.
71    #[serde(rename = "fixed-precision")]
72    FixedPrecision {
73        /// Number of bit planes encoded
74        precision: u32,
75    },
76    /// In fixed-accuracy mode, all transform coefficient bit planes up to a
77    /// minimum bit plane number are encoded. The smallest absolute bit plane
78    /// number is chosen such that
79    /// `$minexp = \text{floor}(\log_{2}(tolerance))$`.
80    #[serde(rename = "fixed-accuracy")]
81    FixedAccuracy {
82        /// Absolute error tolerance
83        tolerance: f64,
84    },
85    /// Lossless per-block compression that preserves integer and floating point
86    /// bit patterns.
87    #[serde(rename = "reversible")]
88    Reversible,
89}

91impl Codec for ZfpCodec {
92    type Error = ZfpCodecError;
93
94    fn encode(&self, data: AnyCowArray) -> Result<AnyArray, Self::Error> {
95        if matches!(data.dtype(), AnyArrayDType::I32 | AnyArrayDType::I64)
96            && matches!(
97                self.mode,
98                ZfpCompressionMode::FixedAccuracy { tolerance: _ }
99            )
100        {
101            return Err(ZfpCodecError::FixedAccuracyModeIntegerData);
102        }
103
104        match data {
105            AnyCowArray::I32(data) => Ok(AnyArray::U8(
106                Array1::from(compress(data.view(), &self.mode)?).into_dyn(),
107            )),
108            AnyCowArray::I64(data) => Ok(AnyArray::U8(
109                Array1::from(compress(data.view(), &self.mode)?).into_dyn(),
110            )),
111            AnyCowArray::F32(data) => Ok(AnyArray::U8(
112                Array1::from(compress(data.view(), &self.mode)?).into_dyn(),
113            )),
114            AnyCowArray::F64(data) => Ok(AnyArray::U8(
115                Array1::from(compress(data.view(), &self.mode)?).into_dyn(),
116            )),
117            encoded => Err(ZfpCodecError::UnsupportedDtype(encoded.dtype())),
118        }
119    }
120
121    fn decode(&self, encoded: AnyCowArray) -> Result<AnyArray, Self::Error> {
122        let AnyCowArray::U8(encoded) = encoded else {
123            return Err(ZfpCodecError::EncodedDataNotBytes {
124                dtype: encoded.dtype(),
125            });
126        };
127
128        if !matches!(encoded.shape(), [_]) {
129            return Err(ZfpCodecError::EncodedDataNotOneDimensional {
130                shape: encoded.shape().to_vec(),
131            });
132        }
133
134        decompress(&AnyCowArray::U8(encoded).as_bytes())
135    }
136
137    fn decode_into(
138        &self,
139        encoded: AnyArrayView,
140        decoded: AnyArrayViewMut,
141    ) -> Result<(), Self::Error> {
142        let AnyArrayView::U8(encoded) = encoded else {
143            return Err(ZfpCodecError::EncodedDataNotBytes {
144                dtype: encoded.dtype(),
145            });
146        };
147
148        if !matches!(encoded.shape(), [_]) {
149            return Err(ZfpCodecError::EncodedDataNotOneDimensional {
150                shape: encoded.shape().to_vec(),
151            });
152        }
153
154        decompress_into(&AnyArrayView::U8(encoded).as_bytes(), decoded)
155    }
156}

172#[derive(Debug, Error)]
173/// Errors that may occur when applying the [`ZfpCodec`].
174pub enum ZfpCodecError {
175    /// [`ZfpCodec`] does not support the dtype
176    #[error("Zfp does not support the dtype {0}")]
177    UnsupportedDtype(AnyArrayDType),
178    /// [`ZfpCodec`] does not support the fixed accuracy mode for integer data
179    #[error("Zfp does not support the fixed accuracy mode for integer data")]
180    FixedAccuracyModeIntegerData,
181    /// [`ZfpCodec`] only supports 1-4 dimensional data
182    #[error("Zfp only supports 1-4 dimensional data but found shape {shape:?}")]
183    ExcessiveDimensionality {
184        /// The unexpected shape of the data
185        shape: Vec<usize>,
186    },
187    /// [`ZfpCodec`] was configured with an invalid expert `mode`
188    #[error("Zfp was configured with an invalid expert mode {mode:?}")]
189    InvalidExpertMode {
190        /// The unexpected compression mode
191        mode: ZfpCompressionMode,
192    },
193    /// [`ZfpCodec`] failed to encode the header
194    #[error("Zfp failed to encode the header")]
195    HeaderEncodeFailed,
196    /// [`ZfpCodec`] failed to encode the data
197    #[error("Zfp failed to encode the data")]
198    ZfpEncodeFailed,
199    /// [`ZfpCodec`] can only decode one-dimensional byte arrays but received
200    /// an array of a different dtype
201    #[error(
202        "Zfp can only decode one-dimensional byte arrays but received an array of dtype {dtype}"
203    )]
204    EncodedDataNotBytes {
205        /// The unexpected dtype of the encoded array
206        dtype: AnyArrayDType,
207    },
208    /// [`ZfpCodec`] can only decode one-dimensional byte arrays but received
209    /// an array of a different shape
210    #[error("Zfp can only decode one-dimensional byte arrays but received a byte array of shape {shape:?}")]
211    EncodedDataNotOneDimensional {
212        /// The unexpected shape of the encoded array
213        shape: Vec<usize>,
214    },
215    /// [`ZfpCodec`] failed to decode the header
216    #[error("Zfp failed to decode the header")]
217    HeaderDecodeFailed,
218    /// [`ZfpCodec`] cannot decode into the provided array
219    #[error("ZfpCodec cannot decode into the provided array")]
220    MismatchedDecodeIntoArray {
221        /// The source of the error
222        #[from]
223        source: AnyArrayAssignError,
224    },
225    /// [`ZfpCodec`] failed to decode the data with the unknown dtype
226    #[error("Zfp failed to decode the data with an unknown dtype #{0}")]
227    DecodeUnknownDtype(u32),
228    /// [`ZfpCodec`] failed to decode the data
229    #[error("Zfp failed to decode the data")]
230    ZfpDecodeFailed,
231}

233/// Compress the `data` array using ZFP with the provided `mode`.
234///
235/// # Errors
236///
237/// Errors with
238/// - [`ZfpCodecError::ExcessiveDimensionality`] if data is more than
239///   4-dimensional
240/// - [`ZfpCodecError::InvalidExpertMode`] if the `mode` has invalid expert mode
241///   parameters
242/// - [`ZfpCodecError::HeaderEncodeFailed`] if encoding the ZFP header failed
243/// - [`ZfpCodecError::ZfpEncodeFailed`] if an opaque encoding error occurred
244pub fn compress<T: ffi::ZfpCompressible, D: Dimension>(
245    data: ArrayView<T, D>,
246    mode: &ZfpCompressionMode,
247) -> Result<Vec<u8>, ZfpCodecError> {
248    // Setup zfp structs to begin compression
249    let field = ffi::ZfpField::new(data)?;
250    let stream = ffi::ZfpCompressionStream::new(&field, mode)?;
251
252    // Allocate space based on the maximum size potentially required by zfp to
253    //  store the compressed array
254    let stream = stream.with_bitstream(field);
255
256    // Write the full header so we can reconstruct the array on decompression
257    let stream = stream.write_full_header()?;
258
259    // Compress the field into the allocated output array
260    stream.compress()
261}

263/// Decompress the `encoded` data into an array using ZFP.
264///
265/// # Errors
266///
267/// Errors with
268/// - [`ZfpCodecError::HeaderDecodeFailed`] if decoding the header failed
269/// - [`ZfpCodecError::DecodeUnknownDtype`] if the encoded data uses an unknown
270///   dtype
271/// - [`ZfpCodecError::ZfpDecodeFailed`] if an opaque decoding error occurred
272pub fn decompress(encoded: &[u8]) -> Result<AnyArray, ZfpCodecError> {
273    // Setup zfp structs to begin decompression
274    let stream = ffi::ZfpDecompressionStream::new(encoded);
275
276    // Read the full header to verify the decompression dtype
277    let stream = stream.read_full_header()?;
278
279    // Decompress the field into a newly allocated output array
280    stream.decompress()
281}

283/// Decompress the `encoded` data into a `decoded` array using ZFP.
284///
285/// # Errors
286///
287/// Errors with
288/// - [`ZfpCodecError::HeaderDecodeFailed`] if decoding the header failed
289/// - [`ZfpCodecError::DecodeUnknownDtype`] if the encoded data uses an unknown
290///   dtype
291/// - [`ZfpCodecError::MismatchedDecodeIntoArray`] if the `decoded` array is of
292///   the wrong dtype or shape
293/// - [`ZfpCodecError::ZfpDecodeFailed`] if an opaque decoding error occurred
294pub fn decompress_into(encoded: &[u8], decoded: AnyArrayViewMut) -> Result<(), ZfpCodecError> {
295    // Setup zfp structs to begin decompression
296    let stream = ffi::ZfpDecompressionStream::new(encoded);
297
298    // Read the full header to verify the decompression dtype
299    let stream = stream.read_full_header()?;
300
301    stream.decompress_into(decoded)
